import maya.cmds as cmds

def MakeCOutliner():

	mainWin = "C.H. Outliner"
	
	sels = cmds.ls (sl = True)
	
	if (cmds.window (mainWin, exists = True)):
		cmds.deleteUI (mainWin, window = True)
		
	#These lines only make the windows
	mainWin = cmds.window (mainWin)
	
	columnObj = cmds.rowColumnLayout (numberOfColumns=2, parent = mainWin)
        cmds.button()
        cmds.button()
	cmds.textScrollList(ams = True, a = (sels))
    	#allowedAreas = ['right', 'left']
        #cmds.dockControl( area='left', content=mainWin, allowedArea=allowedAreas )
	
	cmds.showWindow(mainWin)
	
MakeCOutliner()
